Как получить путь к файлу скрипта VBScript ?

Иногда бывает необходимо определить в какой папке находится запущенный скрипт VBS. В этой заметке показано 3 способа как опеределить путь к запущенному скрипту на VBScript.

Способ 1

?

1

2

Set WshShell = WScript.CreateObject("WScript.Shell")

CurrentPath = WshShell.CurrentDirectory

Способ 2

?

1

2

Set FSO = CreateObject("Scripting.FileSystemObject")

CurrentPath = FSO.GetParentFolderName(WScript.ScriptFullName)

Способ 3

?

1

CurrentPath = Mid(WScript.ScriptFullName,1,InStrRev(WScript.ScriptFullName,"\")-1)
