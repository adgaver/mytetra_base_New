Function fun_Connect_EX_File(ByVal oDb As Database, _

ByVal str_EX_File As String, _

ByVal str_ShName As String) As Boolean

Dim blnTblConnect As Boolean

On Error GoTo Error_fun_Connect_EX_File


Application.StatusBar = "Подключение таблицы " & str_EX_File


'удаление подключенных листов

blnTblConnect = m_Connect_EX_File.fun_ObjExists(oDb:=oDb, strObjType:="t", strObjName:="EX_" & str_ShName) '"Table", "Query",

If blnTblConnect Then oDb.Execute "DROP TABLE " & "EX_" & str_ShName 'удаление если таблица существует


'подключение листов напрямую в базу

blnTblConnect = m_Connect_EX_File.fun_Append_Ex(oDb:=oDb, _

str_File:=str_EX_File, _

str_ShName:=str_ShName, _

str_TblName:="EX_" & str_ShName)


fun_Connect_EX_File = True


Exit_fun_Connect_EX_File:

Application.StatusBar = ""

On Error GoTo 0

Exit Function


Error_fun_Connect_EX_File:

fun_Connect_EX_File = False

MsgBox "Error " & Err.Number & " (" & Err.Description & ") " & vbNewLine & _

"Файл: " & str_EX_File & "не подключен" & vbNewLine & _

"in procedure fun_Append_Ex of Module m_Connect_EX_File"

Resume Exit_fun_Connect_EX_File

End Function

Public Function fun_ObjExists(ByVal oDb As Object, _

ByVal strObjType As String, _

ByVal strObjName As String) As Boolean

On Error Resume Next

oDb.TableDefs.Refresh

Select Case strObjType

Case "t": fun_ObjExists = (oDb.TableDefs(strObjName).Name = strObjName)

Case "z": fun_ObjExists = (oDb.QueryDefs(strObjName).Name = strObjName)

Case Else: fun_ObjExists = (oDb.Containers(strObjType & "s").Documents(strObjName).Name = strObjName)

End Select


Err.Clear

End Function
