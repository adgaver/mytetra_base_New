Public Function fn_DeleteFolder_CMD(ByVal str_path_delete As String, _

ByVal bln_recovery As Boolean) As Boolean


Dim Result As Boolean

Dim exit_Time As Date

Dim bln_path_exists As Boolean

On Error GoTo ErrorHandler

'максимальное время для удаления

exit_Time = Now + CDate("00:00:10")


'проверяем существование папки

bln_path_exists = fn_PathExists(str_path_delete)

Select Case bln_path_exists


Case True:

'https://www.excel-vba.ru/chto-umeet-excel/kak-udalit-papku-ili-vse-fajly-iz-papki-cherez-vba/

Shell "cmd /c rd /S/Q """ & str_path_delete & """", vbHide

bln_path_exists = fn_PathExists(str_path_delete)

Do While bln_path_exists 'пока папка существует

'ожидание, т.к. выполнение CMD немного запаздывает, если не удалена за отведенное время выходим

If exit_Time < Now Then

fn_DeleteFolder_CMD = Result

Exit Function

End If

bln_path_exists = fn_PathExists(str_path_delete)

Loop

'Debug.Print Now, nextTime


Case False:

Result = True


End Select


If bln_recovery Then Call fn_CreateCatalog(str_Path:=str_path_delete)


fn_DeleteFolder_CMD = Result


Exit Function

ErrorHandler:


Select Case Err

'обработка ошибок для использования раскомментировать

'Case

Case Else:

Debug.Print "Произошла ошибка в fn_DeleteFolder_CMD" & vbNewLine & Err.Number & vbNewLine & Err.Description

End Select

End Function

Function fn_PathExists(str_Path) As Boolean

On Error Resume Next

fn_PathExists = GetAttr(str_Path) And vbDirectory = vbDirectory

End Function
