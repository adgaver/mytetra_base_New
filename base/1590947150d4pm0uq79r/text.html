Function rs(sql As String) As ADODB.Recordset
    Set rs = New ADODB.Recordset
    Set rs.ActiveConnection = CurrentProject.Connection
    rs.Open sql
End Function

Function fun_rsADO(ByVal strSQL As String) As ADODB.Recordset
    Set fun_rsADO = New ADODB.Recordset
    Set fun_rsADO.ActiveConnection = CurrentProject.Connection
    fun_rsADO.Open strSQL
End Function
