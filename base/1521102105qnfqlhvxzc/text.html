В следующем примере кода показано, как изменить размер элемента управления с формой. Использует класс OlkControl , представляющий некоторые основные свойства (например, HorizontalLayout и ** VerticalLayout) общие для Outlook элементы управления формы. Он предполагается существующей Outlook текстового поля, myTextBox, в форме и использует приведение в Visual Basic, чтобы разрешить текстового поля для использования свойства **OlkControl.


Dim olkCtrl As Outlook.OlkControl


' Let the text box control use the properties of OlkControl

Set olkCtrl = myTextBox


' Enable automatic adjustments of the layout with respect to the rest of the form

olkCtrl.EnableAutoLayout = True


' Allow resizing the text box control horizontally and vertically with the form

olkCtrl.HorizontalLayout = olHorizontalLayoutGrow

olkCtrl.VerticalLayout = olVerticalLayoutGrow
