На вашем компьютере должен быть установлен win32com. Вот мой код:

import win32com.client as win32
fname = "full+path+to+xls_file"
excel = win32.gencache.EnsureDispatch('Excel.Application')
wb = excel.Workbooks.Open(fname)

wb.SaveAs(fname+"x", FileFormat = 51)    #FileFormat = 51 is for .xlsx extension
wb.Close()                               #FileFormat = 56 is for .xls extension
excel.Application.Quit()

def convert_xls_to_xlsx(fname_xls):# -> None:
    fname_xlsx = os.path.basename(fname_xls) + "x"
    fname_xlsx = os.path.join(r"C:\Users\Public\Documents", fname_xlsx)


    # удаляем файл
    if os.path.exists(fname_xlsx):
        os.remove(fname_xlsx)    


    try: # при удачной конвертации возвращаем путь к новому файлу
        excel = win32.gencache.EnsureDispatch('Excel.Application')
        wb = excel.Workbooks.Open(fname_xls)


        wb.SaveAs(fname_xlsx, FileFormat = 51)    #FileFormat = 51 is for .xlsx extension
        wb.Close()                               #FileFormat = 56 is for .xls extension
        excel.Application.Quit()
        return fname_xlsx
    except:
        return ''


print(convert_xls_to_xlsx(fname))
