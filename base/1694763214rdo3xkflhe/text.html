Const intRowEachStart As Integer = 4 '4-5 строки для поиска

Const intRowEachEnd As Integer = 5 '3-4 строки для поиска

Const int_Row_FldName As Integer = 6 '5 шапка

Const intClnEachStart As Integer = 2 'стартовый столбец для поиска и экспорта


Private Sub Worksheet_Change(ByVal Target As Range)

int_Cln_EachEnd = Cells(int_Row_FldName, 256).End(xlToLeft).Column


str_Address = Range(Cells(intRowEachStart, intClnEachStart), Cells(intRowEachEnd, int_Cln_EachEnd)).Address

'If Not Intersect(Target, Range("B2:H3")) Is Nothing Then

If Not Intersect(Target, Range(str_Address)) Is Nothing Then

On Error Resume Next


'скопировать для запоминания

Range(Target.Address).Copy

Cells(1).PasteSpecial Paste:=xlPasteValues, Operation:=xlNone, SkipBlanks:=False, Transpose:=False


'выполнить поиск и сортировку

Call Me.Run_Filter(Target.Address)


'восстановить текст в буфер обмена

Cells(1).Copy

Range(str_CellAddress).PasteSpecial Paste:=xlPasteValues, Operation:=xlNone, SkipBlanks:=False, Transpose:=False


'вернуться в исходную ячейку

Target.Select

Target.Show

End If

End Sub

'* Sub : Run_Filter - поиск и сортировка столбца по заданному тексту

'* Created : 15-09-2023 10:29

'* Author : -

'* Contacts : -

'* Copyright : -

'* Argument(s): Description

'*

'* ByVal str_CellAddress As String : текст для поиска

'*

'* * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * *

Sub Run_Filter(ByVal str_CellAddress As String)

Dim str_Find As String

Dim int_Counter As Integer 'количество вхождений

Dim i As Long, j As Long, Lng_RowEnd As Long

Dim int_ClnVal As Integer, int_ClnEnd As Integer

Dim aValEach As Variant 'массив значений для поиска

'Dim iStart As Integer: iStart = int_Row_FldName + 1


Dim f1 As Integer, F2 As Integer


int_ClnVal = Range(str_CellAddress).Column

Lng_RowEnd = Columns(intClnEachStart).Rows(Rows.Count).End(xlUp).Row

int_ClnEnd = Cells(int_Row_FldName, 256).End(xlToLeft).Column


aTemp = Range(Cells(int_Row_FldName + 1, int_ClnVal), Cells(Lng_RowEnd, int_ClnVal)).Value 'строки (где ищем)

aValEach = Range(Cells(intRowEachStart, int_ClnVal), Cells(intRowEachEnd, int_ClnVal)).Value 'строки (что ищем)

ReDim aF(LBound(aTemp, 1) To UBound(aTemp, 1), LBound(aTemp, 2) To UBound(aTemp, 2))

'------------------------------------------------

''сброс фильтра

'Фильтр = m_main.FnFiltersSheet(ActiveSheet, int_Row_FldName, 1)

'открыть все строки

With Cells

.EntireRow.Hidden = False

.EntireColumn.Hidden = False

End With


'------------------------------------------------

'очистить цвета

With Range(Cells(int_Row_FldName + 1, 1), Cells(Lng_RowEnd, int_ClnEnd))

With .Font

.ThemeColor = xlThemeColorLight1

.TintAndShade = 0

End With

.Font.Bold = True

.Font.Bold = False

End With


'------------------------------------------------

'проверка значений массива

str_Find = ""

For i = LBound(aValEach) To UBound(aValEach)

'если строка пустая, пытаемся заполнить

If Len(str_Find) = 0 Then str_Find = Trim(aValEach(i, 1))

Next i

If Len(str_Find) = 0 Then

Call Me.EventsChange(True)

Exit Sub

End If


'------------------------------------------------

Call Me.EventsChange(False)


'------------------------------------------------

For i = LBound(aTemp) To UBound(aTemp)

aF(i, 1) = 0 'устанавливаем начальные значения

For j = LBound(aValEach) To UBound(aValEach)

str_Find = Trim(aValEach(j, 1))

If Len(str_Find) > 0 Then

If InStr(1, aTemp(i, 1), str_Find, vbTextCompare) Then

aF(i, 1) = aF(i, 1) + Me.fn_ShowInstr(Cells(i + int_Row_FldName, int_ClnVal), str_Find)

End If

End If

If aF(i, 1) > 0 Then

Debug.Print i

End If

Next j

Next i

Cells(int_Row_FldName + 1, 1).Resize(UBound(aF), UBound(aTemp, 2)) = aF

Cells(int_Row_FldName, 1).Value = "Count"


Call Me.EventsChange(True)


'сортировка по столбце со значениями

Cells(int_Row_FldName + 1, int_ClnVal).Sort Key1:=Cells(int_Row_FldName, int_ClnVal), Order1:=xlAscending, Header:=xlYes

'сортировка по столбцу с количеством

Cells(int_Row_FldName + 1, 1).Sort Key1:=Cells(int_Row_FldName, 1), Order1:=xlDescending, Header:=xlYes


End Sub

'* Function : fn_ShowInstr - Заливка совпавших символов

'* Created : 15-09-2023 10:26

'* Author : adgaver

'* Contacts : -

'* Copyright : -

'* Argument(s): Description

'*

'* ByVal rCell As Range : ячейка для анализа текста

'* ByVal str_Find As String : текст для поиска и заливки

'*

'* * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * *

Function fn_ShowInstr(ByVal rCell As Range, _

ByVal str_Find As String) As Integer

Dim str_Main As String:

Dim iLenFind As Integer, iLenMainFind As Integer

Dim iCh As Integer 'счётчик совпадений


'все в верхний регистр

str_Find = UCase(str_Find)

str_Main = UCase(rCell.Value) '


iLenMainFind = Len(str_Main) 'длина строки

iLenFind = Len(str_Find) 'длина строки поиска


If InStr(1, str_Main, str_Find, vbTextCompare) Then 'если есть в слове, то перебирам все символы


For i = 1 To iLenMainFind - iLenFind + 1

str_FindMain = Mid(str_Main, i, iLenFind)


If str_FindMain = str_Find Then

With rCell

.Characters(Start:=i, Length:=Len(str_Find)).Font.ColorIndex = 3

.Characters(Start:=i, Length:=Len(str_Find)).Font.Bold = True

End With

iCh = iCh + 1

End If

Next i


fn_ShowInstr = iCh

Else

fn_ShowInstr = 0

End If


End Function

' .Calculation = xlCalculationAutomatic

.ScreenUpdating = Value

.ShowWindowsInTaskbar = Value

.DisplayAlerts = Value

.EnableEvents = Value

If Value Then

.Calculation = xlCalculationAutomatic

Else: .Calculation = xlCalculationManual

End If
