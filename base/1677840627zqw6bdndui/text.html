let //https://comrade-xl.ru/2019/08/23/udalit-lishnie-probeli/

   source = Excel.CurrentWorkbook(){[Name = "Таблица1"]}[Content],

   tab_add_col = Table.AddColumn(

      source,

      "ФИО",

      each List.Select(

         Text.Split([160], Character.FromNumber(32))

            & Text.Split([160], Character.FromNumber(160)),

         each _

            <> ""

               and not Text.Contains(_, Character.FromNumber(160))

               and not Text.Contains(_, Character.FromNumber(32))

      )

   ),

   col_expand = Table.TransformColumns(

      tab_add_col,

      {

         "ФИО",

         each Text.Combine(List.Transform(_, Text.From), " "),

         type text

      }

   )

in

   col_expand
