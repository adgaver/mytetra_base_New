    //Конвертация даты со временем 24-00 в Power Query

    //https://stackoverflow.com/questions/68381588/more-than-24-hr-format-issue-in-power-query

    //https://www.planetaexcel.ru/forum/index.php?PAGE_NAME=read&FID=1&TID=133750&TITLE_SEO=133750-konvertatsiya-daty-so-vremenem-2400-v-power-query&PAGE_NA=&tags=averagex

    fun_Transform_Time_24 = (this as text) => 

                            let

                                hours = Number.FromText(Text.BeforeDelimiter(this, ":"))

                            in

                                #duration(Number.IntegerDivide(hours,24), Number.Mod(hours,24),0,0)

                                Duration.FromText("0.00:" & Text.AfterDelimiter(this,":")),

    Transform_Time =Table.TransformColumns(old_step,

                                            {

                                            {"Время", fun_Transform_Time_24}

                                            }),

    Changed_Type = Table.TransformColumnTypes(Transform_Time,{

                        {"Время", type duration}})
