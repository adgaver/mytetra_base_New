Function fun_Concat_ar(ByVal int_Cln_ar As Integer)

Const CONST_int_RowStart As Integer = 7

Const CONST_int_RowEnd As Integer = 20

Dim ar As Variant, strLink As String

ar = Range(Cells(CONST_int_RowStart, int_Cln_ar), Cells(CONST_int_RowEnd, int_Cln_ar)).Value

strLink = ar(1, 1)

For i = LBound(ar) + 1 To UBound(ar)

strLink = strLink & ar(i, 1)

Next i

fun_Concat_ar = Trim(strLink)

End Function
