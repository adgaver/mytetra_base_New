import os

import pandas as pd


# Этот метод использует XLS к XLSX

def transformat():

global path # Настраивается как глобальная переменная

path = os.getcwd()# Получить текущий рабочий путь

file = os.listdir(path)# Получить все файлы под текущим путем

for f in file:

file_name_be,suff = os.path.splitext(f)# Сегментировать путь, соответственно, путь к файлу и файловый суффикс

if suff == '.xls':

print(«Конвертировать {} файлы ...».format(f))

data = pd.DataFrame(pd.read_excel(path + '\\' + f))# Прочитайте файл XLS

data.to_excel(path + '\\' + file_name_be + «Формат преобразования. XLSX»,index = False)#Format преобразование

print(Файл '{} был преобразован в {} Сохранить \ n' в каталоге {}.format(f,file_name_be + «Формат преобразования. XLSX»,path))
