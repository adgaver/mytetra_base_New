'* Function : fn_each_position_in_Range - поиск значения в диапазоне, вернет ячейку (rCell)

'* Created : 27-09-2023 11:12

'* Author : girevav

'* Copyright : -

'* Argument(s): Description

'*

'* ByVal rRange As Range : диапазон в котором ищем

'* str_each As String : строка для поиска

'* bln_accord As Boolean : точность поиска True=полное соответствие

'* Set rCell = fn_each_position_in_Range(rRange:=Range(.Cells(1, 1), .Cells(1, Lng_Cln_End)), str_each:="ТЕКСТ ДЛЯ ПОИСКА", bln_accord:=True)

'* If TypeName(rCell) = "Range" Then Lng_Cln_Each = rCell.Column

'* * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * *

Function fn_each_position_in_Range( _

ByVal rRange As Range, _

ByVal str_each As String, _

ByVal bln_accord As Boolean) As Range

Dim rCell As Range

If bln_accord Then

For Each rCell In rRange.Cells

If rCell.Value = str_each Then

Set fn_each_position_in_Range = rCell

Exit Function

End If

Next [rCell]

Else

For Each rCell In rRange.Cells

If rCell.Value Like "*" & str_each & "*" Then

Set fn_each_position_in_Range = rCell

Exit Function

End If

Next [rCell]

End If

End Function
