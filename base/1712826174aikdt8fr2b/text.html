REM Добавить путь "C:\ffmpeg\bin" в переменные среды


REM 133MB->24MB
ffmpeg -i input.mp4 -vcodec h264 -acodec mp2 output_h264_mp2.mp4


REM 133MB->7MB
ffmpeg -i input.mp4  -vcodec libx265 -crf 28 output_libx265.mp4


REM 133MB->42MB (хрен угадаешь с размером)
ffmpeg -i input.mp4 -s 1920x1080 -acodec copy -y  output_acodec.mp4


REM 133MB->16MB
ffmpeg -i input.mp4 -vcodec h264 -b:v 1000k -acodec mp3 output_h264_mp3_1000k.mp4


REM 133MB->12MB
ffmpeg -i input.mp4 -vcodec h264 -b:v 700k -acodec mp3 output_h264_mp3_700k.mp4


REM и если вы не хотите звук, попробуйте:
ffmpeg -i input.mp4 -vcodec h264 -b:v 700k -an output_h264_mp3_700k_без_звука.mp4


REM 133MB->19MB
ffmpeg -i input.mp4 output.mkv


REM 133MB->7MB не переворачивает
ffmpeg -i input.mp4 -vf scale=640:360 output_scale_360_640.mp4
