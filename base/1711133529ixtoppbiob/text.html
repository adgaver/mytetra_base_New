% Процентов (Итог) = 
var pr_Table = 
    CALCULATETABLE (
        ADDCOLUMNS ( 
            SUMMARIZE ( 
                CALCULATETABLE( --такая конструкция, если нужны условия отбора. Если не нужны, то можно просто таблицу прописать
                    DISTINCT('table'[column]),
                    'table'[column] <> "0" &&  'table'[column] <> BLANK()
                ), 
                'table'[column] 
            ), 
            "@perc", 
            [%NC] -- мера расчета процентов
        ),
        ALL ( 'table' )
    )
RETURN
DIVIDE(
    SUMX(pr_Table,[@perc]),
    COUNTROWS (pr_Table),
    0
)
