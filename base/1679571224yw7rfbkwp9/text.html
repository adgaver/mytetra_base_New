Sub Delete_Connections(ByVal oWb As Workbook)

    'удалить подключения

    For Each oCon In oWb.Connections

        oCon.Delete

    Next [oCon]

End Sub

'------------------------------------------------------------------------------

Sub Delete_Shape(ByVal oWb As Workbook)

    'удалить кнопки

    For Each oSh In oWb.Sheets

        For Each oShape In oSh.Shapes

            oShape.Delete

        Next [oShape]

    Next [oSh]

End Sub

'------------------------------------------------------------------------------

Sub Delete_Lnk(ByVal oWb As Workbook)

    ' Удалить все связи

    With oWb

        For Each Lnk In .LinkSources(Type:=xlLinkTypeExcelLinks)

            .BreakLink Lnk, xlLinkTypeExcelLinks

        Next [Lnk]

    End With

End Sub
