//Punctuations

= {" ", "~", ",", ".", "?", "!", "(", ")", "[", "]", "{", "}", "@",

"#", "$", "%", "^", "&", "*", "-", "+", "=", ":", ";", "|",

"<", ">", "/", """, """, "\", """", "#(tab)", "#(cr)", "#(lf)"}


//----------------------------------------------------------------------------- 

let

str_Text_Split = List.Accumulate(

        {" ", "~", ",", ".", "?", "!", "(", ")", "[", "]", "{", "}", "@",

        "#", "$", "%", "^", "&", "*", "-", "+", "=", ":", ";", "|",

        "<", ">", "/", """, """, "\", """", "#(tab)", "#(cr)", "#(lf)"}

    Источник = Excel.CurrentWorkbook(){[Name="Таблица1"]}[Content],

    #"Сортированные строки" = Table.Sort(Источник,{{"Nb", Order.Ascending}}),

 

    //удаляем двойные пробелы и переносы строк

    str_Text_Split=Text.From(Character.FromNumber(32)&Character.FromNumber(160) ),

    fun_Clear_text = (this as text) as text => Text.Combine(

                                          List.Select(

                                                      Text.SplitAny(this, str_Text_Split),each _ <> ""), " "),

   

    Clear_text = Table.TransformColumns( #"Сортированные строки"  ,

                                        {

                                        {"1", fun_Clear_text, type text}

                                        })

in

    Clear_text

let

    Источник = Excel.CurrentWorkbook(){[Name="Таблица1"]}[Content],

    #"Сортированные строки" = Table.Sort(Источник,{{"Nb", Order.Ascending}}),

 

    //удаляем двойные пробелы и переносы строк

    str_Text_Split=Text.From(Character.FromNumber(32)&Character.FromNumber(160) & "#(lf)" ),

    fun_Clear_text = (this as text) as text => Text.Combine(

                                          List.Select(

                                                      Text.SplitAny(this, str_Text_Split),each _ <> ""), " "),

   

    Clear_text = Table.TransformColumns( #"Сортированные строки"  ,

                                        {

                                        {"1", fun_Clear_text, type text}

                                        })

in

    Clear_text
