def exit_program(str_Title=str_Title, str_Msg=str_Msg, x=350, y=100):

# чтение в одну строку
def read_file_ALL(str_File=r'path_to_file.txt', str_encoding=""):
    if not os.path.exists(str_File):
        # если файла нет, то создадим пустой
        with open(str_File, 'w') as f :
            f.write('')
            f.close()
            return ''
    else:
        if len(str_encoding) == 0:
            with open(str_File) as f:
                s = f.read()
                # if not f.closed: 
                #     f.close()
                return s
        elif str_encoding == "utf8": 
            with open(str_File, encoding="utf8") as f:
                s = f.read()
                return s
        elif str_encoding == "ansi": 
            with open(str_File, encoding="ansi") as f:
                s = f.read()
                return s
        f.close()



# построчное чтение    
def read_file_Line(str_File=r'path_to_file.txt', str_encoding=""):
    # проверка существования файла
    check_file = os.path.exists(str_File) # True
    if check_file:
        # открытие и чтение файла
        
        with open(str_File, encoding="utf8") as file:
            lines = [line.rstrip() for line in file]
            # проверка длины списка
            if len(lines) > 0 :
                return lines
            else:
                str_Title = 'Ошибка чтения файла!'
                str_Msg = 'В файле не найдено записей!'
                exit_program(str_Title, str_Msg, 350, 70)
        file.close()
    else:
        str_Title = 'Ошибка открытия файла!'
        str_Msg = 'Текстовый файл не найден!\n\n' + str_File
        exit_program(str_Title, str_Msg, 350, 95)

def fn_Each_Element(driver, 
                    int_Time_wait, 
                    str_Each_method, 
                    str_Each_Element, 
                    str_Each_Element_Title):
    try:
        if (str_Each_method == "NAME"):
            obg_Each_Element = WebDriverWait(driver, int_Time_wait).until(
                EC.element_to_be_clickable((By.NAME, str_Each_Element))
            )


        if (str_Each_method == "XPATH"):
            obg_Each_Element = WebDriverWait(driver, int_Time_wait).until(
                EC.element_to_be_clickable((By.XPATH, str_Each_Element))
            )


        if (str_Each_method == "ID"):
            obg_Each_Element = WebDriverWait(driver, int_Time_wait).until(
                EC.element_to_be_clickable((By.ID, str_Each_Element))
            )


        if (str_Each_method == "CLASS_NAME"):
            obg_Each_Element = WebDriverWait(driver, int_Time_wait).until(
                EC.element_to_be_clickable((By.CLASS_NAME, str_Each_Element))
            )


        # print(f"========= {datetime.datetime.now().time()} {str_Each_Element_Title}")    
        return obg_Each_Element
    except:
                print(f"Элемент {str_Each_Element_Title} не найден!!")
                exit_program('Ошибка поиска элемента!', 
                             f"\nЭлемент {str_Each_Element_Title} не найден!!", 
                             350, 
                             75)
