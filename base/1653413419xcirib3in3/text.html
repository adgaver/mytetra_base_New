//-----------------------------------------------------------------------------

Рейтинг Город = RANKX( ALL ( 'Данные'[Город] ), CALCULATE( SUM([Итого] )),,DESC)

Рейтинг Заказчик = RANKX( ALL ( 'Данные'[Заказчик] ), CALCULATE( SUM([Итого])),,DESC)


//-----------------------------------------------------------------------------

Продажи = SUM( 'продажи'[Продажи] )

Всего продажи = CALCULATE( [Продажи] ; ALL( 'номенклатура' ) )

Доля, % от продаж = DIVIDE( [Продажи] ; [Всего продажи] )

Доля, % Город = DIVIDE(SUM( 'Данные'[Итого]) , CALCULATE(SUM( 'Данные'[Итого] ), ALL('Данные'[Город])))

Доля, % Заказчик = DIVIDE(SUM( 'Данные'[Итого]) , CALCULATE(SUM( 'Данные'[Итого] ), ALL('Данные'[Заказчик])))
