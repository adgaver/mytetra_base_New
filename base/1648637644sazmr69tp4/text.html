    str_Text_Split=Text.From(Character.FromNumber(32)&Character.FromNumber(160)&"."),

    fun_Clear_text = (this as text) as text => Text.Combine(List.Select(Text.SplitAny(this, str_Text_Split),each _ <> ""), " "),

   

    Clear_text = Table.TransformColumns(old_step,

                                        {

                                        {"Column4", fun_Clear_text, type text},

                                        {"Column5", fun_Clear_text, type text},

                                        {"Column6", fun_Clear_text, type text},

                                        {"Column7", fun_Clear_text, type text},

                                        {"Column8", fun_Clear_text, type text},

                                        {"Column9", fun_Clear_text, type text},

                                        {"Column10", fun_Clear_text, type text}

                                        }),

                                        

    #"Повышенные заголовки" = Table.PromoteHeaders(Clear_text, [PromoteAllScalars=true]),
