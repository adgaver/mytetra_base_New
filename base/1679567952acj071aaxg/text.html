let

    Source = List.Dates( Start, Number.From( End - Start) +1, #duration(1,0,0,0)),

    Custom1 = List.Select(Source, (_)=>Date.DayOfWeek(_, Day.Monday) = 0),

    #"Calculated Count" = List.NonNullCount(Custom1)

in

    #"Calculated Count"

   

let

    Source = (Start as date, End as date) => let

        Source = List.Dates( Start, Number.From( End - Start) +1, #duration(1,0,0,0)),

        Custom1 = List.Select(Source, (_)=>Date.DayOfWeek(_, Day.Monday) = 0),

        #"Calculated Count" = List.NonNullCount(Custom1)

    in

        #"Calculated Count"

in

    Source

 

//-----------------------------------------------------------------------------

let

    Source = (Start , End ) =>

    let

        Source_Calculated =

            if Start = null or  End = null then

                null

            else

                List.NonNullCount(

                                List.Select(

                                            List.Dates( DateTime.Date(Start),

                                                        Number.From( DateTime.Date(End) - DateTime.Date(Start)) + 1,

                                                        #duration(1,0,0,0)) , 

                                            (_)=>Date.DayOfWeek(_, Day.Monday) < 5

                                            )

                                )

    in

        Source_Calculated

in

    Source
