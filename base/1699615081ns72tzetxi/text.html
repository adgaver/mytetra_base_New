Теги:  python excel данные

import os

import xlrd

import xlwt


# Файл, где находится каталог

rootdir = "."

list = os.listdir(rootdir)

# Создание выходных файлов

book=xlwt.Workbook(encoding="utf-8",style_compression=0)

sheet = book.add_sheet('test01')


# Индекс линии выходного файла

outrow = 0


#

for i in range(0, len(list)):

# Судя по файлу фильтра суффикса

path = os.path.join(rootdir, list[i])

suffix = path.split(".")[-1]

if suffix == "xls" or suffix == "xlsx" :

# Откройте файл и получите все имена таблиц

workbook=xlrd.open_workbook(path)

names=workbook.sheet_names()

# Все таблицы

for name in names:

#Colon объект листа через имя листа

worksheet=workbook.sheet_by_name(name)

# Получите номер ряда

nrows=worksheet.nrows

ncols=worksheet.ncols

# Каждая строка

for nrow in range(nrows):

# Прочтите все столбцы текущей строки

for ncol in range(ncols):

# Получить данные устройства текущего номера ранга, напишите выходной файл

data = worksheet.row_values(nrow)[ncol]

sheet.write(outrow,ncol,data)

# Накопление номера строки выходного файла

outrow += 1


# Сохраните на жестком диске

book.save('test1.xls')
