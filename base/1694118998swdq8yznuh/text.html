Свойство Areas объекта Range

Range.Areas — это свойство, возвращающее коллекцию смежных областей (диапазонов) исходного объекта Range, состоящего из нескольких областей.

Свойство Areas смежного диапазона возвращает коллекцию, состоящую из одного исходного диапазона.

Sub Primer()

Dim s As String, myArea As Range

 

'Отображаем адрес выделенного диапазона на активном листе

MsgBox "Адрес выделенного диапазона:" & vbNewLine & Selection.Address

 

'Отображаем адреса смежных областей выделенного диапазона

s = "Адреса смежных областей выделенного диапазона:"

    For Each myArea In Selection.Areas

        s = s & vbNewLine & myArea.Address

    Next

MsgBox s

 

End Sub
