Private Type BROWSEINFO

    hwndOwner As Long
    pidlRoot As Long
    pszDisplayName As Long
    lpszTitle As Long
    ulFlags As Long
    lpfn As Long
    lParam As Long
    iImage As Long
End Type

Private Const BIF_RETURNONLYFSDIRS As Long = &H1 'For finding a folder to start document searching
Private Const BIF_STATUSTEXT As Long = &H4      'Top of the dialog has 2 lines of text for BROWSEINFO.lpszTitle and one line if
Private Const BIF_EDITBOX As Long = &H10        'Add an editbox to the dialog
Private Const BIF_NEWDIALOGSTYLE As Long = &H40 'Use the new dialog layout with the ability to resize
Private Const BIF_USENEWUI As Long = BIF_NEWDIALOGSTYLE Or BIF_EDITBOX
Private Const BIF_SHAREABLE As Long = &H8000    'sharable resources displayed
Private Const MAX_PATH As Long = 260

Private Declare Function CoInitialize Lib "ole32" (ByVal pvReserved As Long) As Long
Private Declare Function SHBrowseForFolder Lib "shell32" Alias "SHBrowseForFolderA" _
    (lpbi As BROWSEINFO) As Long
Private Declare Function SHGetPathFromIDList Lib "shell32" Alias "SHGetPathFromIDListA" _
    (ByVal pidl As Long, ByVal pszPath As Long) As Integer
Private Declare Sub CoTaskMemFree Lib "ole32.dll" (ByVal hMem As Long)
       
Private m_bi As BROWSEINFO
Private m_b(MAX_PATH) As Byte, m_ttl(MAX_PATH) As Byte

Private Sub Class_Initialize()
    CoInitialize 0
    ZeroMemory m_b
    ZeroMemory m_ttl
End Sub
Private Sub Class_Terminate()
    If m_bi.pidlRoot <> 0 Then CoTaskMemFree m_bi.pidlRoot
End Sub

Private Sub ZeroMemory(barr As Variant)
    Dim i As Integer
    For i = LBound(barr) To UBound(barr)
        barr(i) = 0
    Next i
End Sub

Public Property Get DialogTitle() As String
    DialogTitle = Convert(m_ttl)
End Property
Public Property Let DialogTitle(ByVal vNewValue As String)
    Dim i As Integer
    If Len(vNewValue) < MAX_PATH Then
        ZeroMemory m_ttl
        For i = 1 To Len(vNewValue)
            m_ttl(i - 1) = Asc(Mid(vNewValue, i, 1))
        Next i
    End If
End Property

Private Function Convert(barr As Variant) As String
    Dim i As Integer
    i = LBound(barr)
    Convert = ""
    While barr(i) <> 0
        Convert = Convert & Chr(barr(i))
        i = i + 1
    Wend
End Function

Public Function ChosenDir(Optional title As String = "") As String
    Dim l As Long
    If title <> "" Then DialogTitle = title
    ChosenDir = ""
    If m_bi.pidlRoot <> 0 Then CoTaskMemFree m_bi.pidlRoot
    m_bi.hwndOwner = Application.hWndAccessApp
    m_bi.pidlRoot = 0
    m_bi.pszDisplayName = VarPtr(m_b(0))
    m_bi.lpszTitle = VarPtr(m_ttl(0))
    m_bi.ulFlags = BIF_SHAREABLE Or BIF_USENEWUI ' Or BIF_RETURNONLYFSDIRS
    m_bi.lpfn = 0
    m_bi.lParam = 0
    m_bi.iImage = 0
    l = SHBrowseForFolder(m_bi)
    If SHGetPathFromIDList(l, VarPtr(m_b(0))) Then ChosenDir = Convert(m_b)
    CoTaskMemFree l
End Function
