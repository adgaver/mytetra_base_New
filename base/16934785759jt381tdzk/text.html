Есть много способов удалить файл: - простой метод VBA Kill, использование FileSystemObject и т. д. Вот один фрагмент, который мой друг Девакоттай ПанеерСелвам использовал для удаления файлов (если можно так сказать, временно). Преимущество здесь в том, что удаленный файл доступен в корзине, и пользователь может восстановить его при необходимости.

 

 

Private Type SHFILEOPSTRUCT

    hwnd                      As Long

    wFunc                     As Long

    sFrom                     As String

    sTo                       As String

    fFlags                    As Integer

    fAnyOperationsAborted     As Boolean

    hNameMappings             As Long

    lpszProgressTitle         As String

End Type

 

 

#If Win64 Then

' Игнорируйте то, что отладчик выделяет цветом 5 строк ниже, это декларации для 64-битной версии

Private Declare PtrSafe Function SHFileOperation Lib "shell32.dll" Alias "SHFileOperationA" _

        (ByRef lpFileOp As SHFILEOPSTRUCT) As Long

#Else

Private Declare Function SHFileOperation Lib "shell32.dll" Alias "SHFileOperationA" _

        (ByRef lpFileOp As SHFILEOPSTRUCT) As Long

#End If

 

Private Const FO_DELETE = &H3

Private Const FOF_SILENT = &H4

Private Const FOF_NOCONFIRMATION = &H10

Private Const FOF_ALLOWUNDO = &H40

 

 

Sub DeleteFileUsingAPI()

    'Delete Files using VBA, Temporary Deletion using VBA, VBA Code to send files to Recycle Bin

 

    Dim oFilAPI               As SHFILEOPSTRUCT

    Dim lReturn               As Long

    Dim sFile                 As String    ' File that needs to be deleted

 

 

    sFile = "C:\ Temp\VBADUD\Hints and Tips.htm"

 

    With oFilAPI

        .wFunc = FO_DELETE

        .sFrom = sFile

        .sTo = vbNullChar

        .fFlags = FOF_SILENT + FOF_NOCONFIRMATION + FOF_ALLOWUNDO

    End With

 

    ' Use WinAPI User Defined Function

    lReturn = SHFileOperation(oFilAPI)

 

Err_Delete:

    MsgBox (Err.Number & " - " & Err.Description)

    Err.Clear

    Resume Next

End Sub
