'----------------------------------------------------------------------

' ##### перевернуть массив

'

'----------------------------------------------------------------------

Function fn_TransposeArr(ByVal arrSourse As Variant) As Variant

Dim i As Long, j As Long


ReDim Arr(LBound(arrSourse, 2) To UBound(arrSourse, 2), LBound(arrSourse, 1) To UBound(arrSourse, 1))

For i = LBound(arrSourse, 1) To UBound(arrSourse, 1) ' Определяем сколько колонок*

For j = LBound(arrSourse, 2) To UBound(arrSourse, 2) ' Определяем сколько колонок*

Arr(j, i) = UCase(arrSourse(i, j))

Next j

Next i

fn_TransposeArr = Arr

End Function
