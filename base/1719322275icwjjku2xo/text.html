Мера =

Var dayStat =

SUMMARIZE(

'Таблица',

'Таблица'[Дата],

"day sum",

SUM('Таблица'[Факт])

)

Return

MAXX(dayStat, [day sum])

Мера 2 =

MAXX (

VALUES ( 'Таблица'[Дата] ),

CALCULATE (

SUM ( 'Таблица'[Факт] )

)

)


//-----------------------------------------------------------------------------

Мера 3 =

MAXX (

ADDCOLUMNS (

VALUES ( 'Таблица'[Дата] ),

"day sum", CALCULATE ( SUM ( 'Таблица'[Факт] ) )

),

[day sum]

)
