'con, Shapes, Links, Names
Sub delete_ALL_Con(ByVal oWb As Workbook)
    On Error Resume Next
    ' Удалить все подключения
    Dim pq                    As Object
    For Each pq In oWb.Queries
        pq.Delete
    Next
End Sub
'==============================================================================
Sub delete_ALL_Shapes(ByVal oWb As Workbook)
    On Error Resume Next
    ' удалить все кнопки
    For Each oSh In oWb.Sheets
        For Each oShape In oSh.Shapes
            oShape.Delete
        Next [oShape]
    Next [oSh]
End Sub
'==============================================================================
Sub delete_ALL_Links(ByVal oWb As Workbook)
    On Error Resume Next
    ' Удалить все связи
    With oWb
        For Each Lnk In .LinkSources(Type:=xlLinkTypeExcelLinks)
            .BreakLink Lnk, xlLinkTypeExcelLinks
        Next
    End With
End Sub
'==============================================================================
Sub delete_ALL_Names(ByVal oWb As Workbook)
    On Error Resume Next
    ' Удалить все имена
    For Each n In oWb.Names
        n.Delete
    Next n
End Sub
