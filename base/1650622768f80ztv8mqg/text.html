let

    //путь к файлу

    str_File = "Путь к файлу",

    //расширение файла (в верхнем ргистре)

    ext = Text.Upper(

                    List.Last(

                                    Text.Split(str_File, ".")

                    )),

    //выбор способа извлечения данных, в зависимости от расширния

    Source = if ext="XLSX" then

                //Workbook 

                Excel.Workbook(File.Contents(str_File), null, true)

            else

                //txt

                Csv.Document(File.Contents(str_File),[Delimiter="   ",

Columns=255,

Encoding=1251, QuoteStyle=QuoteStyle.None]),

    //удаление пустых столбцов

    Remove_Empty_Columns = fun_Remove_Empty_Columns( Source )

in

    Remove_Empty_Columns//Source

let

    Источник = ( str_File as text) =>

    let

        ////путь к файлу

        //str_File = "Путь к файлу",

        //расширение файла (в верхнем ргистре)

        ext = Text.Upper(

                        List.Last(

                                        Text.Split(str_File, ".")

                        )),

        //выбор способа извлечения данных, в зависимости от расширния

        Source =

            if ext = "XLSX" then

                //Workbook 

                Excel.Workbook(File.Contents(str_File), null, true)

            else

                //txt

                Csv.Document(File.Contents(str_File),[Delimiter="   ",

                                                //возможно можно сократить количество загружаемых столбцов

                                                Columns=255,

                                                Encoding=1251, QuoteStyle=QuoteStyle.None]),

       

        // список имен столбцов

        List_Cln_Name =

            if ext = "XLSX" then

                Table.ColumnNames(Source{0}[Data])

            else

                Table.ColumnNames( Source ),

 

        Prepare_Table  =

            if ext = "XLSX" then

                Table.PromoteHeaders(

                    Table.SelectColumns(

                                // разворачивем все столбцы

                                Table.ExpandTableColumn( Source, "Data",

                                            List_Cln_Name,

                                            List_Cln_Name ),

                                // удаляем все столбцы, кроме развернутых

                                List_Cln_Name ),

                                // повышаем заголовки

                                [PromoteAllScalars=true] )

            else

                Table.PromoteHeaders(

                            // удаляем пустые столбцы

                            fun_Remove_Empty_Columns( Source ),

                            // повышаем заголовки

                            [PromoteAllScalars=true])

    in

        Prepare_Table

in

    Источник
