(start as date, end as date) =>

  let

    days = List.Dates(start, Duration.Days(end-start), #duration(1,0,0,0)),

    holidays = {"01-01", "01-02", "01-03", "01-04", "01-05", "01-06", "01-07", "01-08", "01-09", "01-10", "01-11", "01-12", "01-13", "01-14", "01-15", "02-23", "03-08", "05-01", "05-09", "06-12", "06-22", "11-04"},

    workdays = List.Select(days, each not(List.Contains(holidays, Text.From(_, "dd-MM"))) and (Date.DayOfWeek(_) < 5)),

    result = List.Count(workdays)

  in

    result
